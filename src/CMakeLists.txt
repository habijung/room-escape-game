# Set resource files
file(GLOB_RECURSE RESOURCE_OBJECTS ${CMAKE_CURRENT_SOURCE_DIR}/resource/object/*.obj)
file(GLOB_RECURSE RESOURCE_IMAGES ${CMAKE_CURRENT_SOURCE_DIR}/resource/img/*.bmp)

# Copy files
file(COPY ${RESOURCE_OBJECTS} DESTINATION ${CMAKE_BINARY_DIR}/${CMAKE_BUILD_TYPE}/resource/object)
file(COPY ${RESOURCE_IMAGES} DESTINATION ${CMAKE_BINARY_DIR}/${CMAKE_BUILD_TYPE}/resource/img)
